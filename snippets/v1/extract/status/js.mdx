```js Node
import FirecrawlApp from "@mendable/firecrawl-js";

const app = new FirecrawlApp({
  apiKey: "fc-YOUR_API_KEY"
});

// Step 1: Start an extraction job (returns immediately with job ID)
const extractJob = await app.asyncExtract([
  'https://docs.firecrawl.dev/*', 
  'https://firecrawl.dev/'
], {
  prompt: "Extract the company mission and features from these pages."
});

console.log(`Job started with ID: ${extractJob.id}`);

// Step 2: Poll for status until completion
while (true) {
  const jobStatus = await app.getExtractStatus(extractJob.id);
  console.log(`Status: ${jobStatus.status}`);
  
  if (jobStatus.status === 'completed') {
    console.log("Extraction completed!");
    console.log(jobStatus.data);
    break;
  } else if (jobStatus.status === 'failed' || jobStatus.status === 'cancelled') {
    console.log(`Extraction ${jobStatus.status}: ${jobStatus.error}`);
    break;
  }
  
  await new Promise(resolve => setTimeout(resolve, 2000)); // Wait before next poll
}
```
