```bash cURL
# Step 1: Start the extraction (returns job ID)
curl -X POST https://api.firecrawl.dev/v1/extract \
    -H 'Content-Type: application/json' \
    -H 'Authorization: Bearer YOUR_API_KEY' \
    -d '{
      "urls": ["https://nextbase.com/dash-cams/622gw-dash-cam"],
      "prompt": "Extract details about the best dash cams including prices, features, pros/cons, and reviews.",
      "enableWebSearch": true
    }'

# Step 2: Get the results using the job ID from step 1
curl -X GET https://api.firecrawl.dev/v1/extract/<JOB_ID> \
    -H 'Content-Type: application/json' \
    -H 'Authorization: Bearer YOUR_API_KEY'
```
