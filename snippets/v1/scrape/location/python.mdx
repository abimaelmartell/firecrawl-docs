```python Python
from firecrawl import FirecrawlApp, ScrapeOptions

app = FirecrawlApp(api_key="fc-YOUR_API_KEY")

# Scrape a website:
scrape_result = app.scrape_url('airbnb.com', 
    scrape_options=ScrapeOptions(
        formats=['markdown', 'html'], 
        location={
            'country': 'BR',
            'languages': ['pt-BR']
        }
    )
)
print(scrape_result)
```
