```python
import asyncio
from firecrawl import AsyncFirecrawl
import os

async def main():
    client = AsyncFirecrawl(api_key=os.getenv("FIRECRAWL_API_KEY"))

    schema = {
        "type": "object",
        "properties": {"title": {"type": "string"}},
        "required": ["title"],
    }

    res = await client.extract(
        urls=["https://docs.firecrawl.dev"],
        prompt="Extract title",
        schema=schema,
        poll_interval=1,
        timeout=60,
    )
    print(res.get("status") or res.get("success"))

asyncio.run(main())
```
