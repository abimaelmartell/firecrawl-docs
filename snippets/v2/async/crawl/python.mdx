```python
import asyncio
from firecrawl import AsyncFirecrawl
import os

async def main():
    client = AsyncFirecrawl(api_key=os.getenv("FIRECRAWL_API_KEY"))
    job = await client.crawl("https://docs.firecrawl.dev", limit=5, poll_interval=1, timeout=120)
    print(job.status)

asyncio.run(main())
```
