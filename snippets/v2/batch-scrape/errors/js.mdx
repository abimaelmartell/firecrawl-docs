```js Node
import Firecrawl from '@mendable/firecrawl-js';

const client = new Firecrawl({ apiKey: process.env.FIRECRAWL_API_KEY });

const { id } = await client.startBatchScrape(['https://bad.invalid', 'https://example.com']);
const errors = await client.getBatchScrapeErrors(id);
console.log(errors.errors, errors.robotsBlocked);
```
