```js Node
import Firecrawl from '@mendable/firecrawl-js';

const client = new Firecrawl({ apiKey: process.env.FIRECRAWL_API_KEY });

// Start a batch scrape job
const { id } = await client.startBatchScrape([
  'https://firecrawl.dev',
  'https://docs.firecrawl.dev'
], {
  options: { formats: ['markdown'] },
});

// Wait for completion
const job = await client.batchScrape([
  'https://firecrawl.dev',
  'https://docs.firecrawl.dev'
], { options: { formats: ['markdown'] }, pollInterval: 2, timeout: 120 });

console.log(job.status, job.completed, job.total);
```
