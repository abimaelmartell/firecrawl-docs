```python
from firecrawl import Firecrawl
import os

client = Firecrawl(api_key=os.getenv("FIRECRAWL_API_KEY"))

# Start and wait for crawl
job = client.crawl(
    url="https://docs.firecrawl.dev",
    limit=10,
    poll_interval=2,
    timeout=120,
)
print(job.status, job.completed, job.total)
```
