```js Node
import Firecrawl from '@mendable/firecrawl-js';

const client = new Firecrawl({ apiKey: process.env.FIRECRAWL_API_KEY });

// Start a crawl
const { id } = await client.startCrawl('https://docs.firecrawl.dev', { limit: 10 });

// Wait for completion
const job = await client.crawl('https://docs.firecrawl.dev', { limit: 10, pollInterval: 2, timeout: 120 });
console.log(job.status, job.completed, job.total);
```
