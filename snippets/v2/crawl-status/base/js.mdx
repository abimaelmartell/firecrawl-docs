```js Node
import Firecrawl from '@mendable/firecrawl-js';

const client = new Firecrawl({ apiKey: process.env.FIRECRAWL_API_KEY });

const { id } = await client.startCrawl('https://docs.firecrawl.dev', { limit: 5 });
const status = await client.getCrawlStatus(id);
console.log(status.status, status.completed, status.total);
```
