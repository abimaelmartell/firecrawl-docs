```js Node
import Firecrawl from '@mendable/firecrawl-js';

const client = new Firecrawl({ apiKey: process.env.FIRECRAWL_API_KEY });

// Start a crawl
const { id } = await client.startCrawl('https://docs.firecrawl.dev', { limit: 10 });

// Cancel it
const ok = await client.cancelCrawl(id);
console.log('cancelled:', ok);
```
