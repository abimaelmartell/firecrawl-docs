```js Node
import FirecrawlApp from "@mendable/firecrawl-js";
import { z } from "zod";

const app = new FirecrawlApp({
  apiKey: "fc-YOUR_API_KEY"
});

const eventLinksSchema = z.object({
  article: z.string(),
  tickets: z.string()
});

const pageEventSchema = z.object({
  date: z.string(),
  day: z.string(),
  title: z.string(),
  performers: z.string(),
  works: z.string(),
  links: eventLinksSchema
});

const pageEventListSchema = z.object({
  events: z.array(pageEventSchema)
});

const result = await app.scrape("https://www.gemeinde-gruenwald.de/kultur/terminkalender/index.html", {
  formats: [{
    type: "json",
    schema: pageEventListSchema
  }],
  includeTags: [".search-form__resultlist"]
});

console.log(result);
```
