```python Python
from firecrawl import Firecrawl
from pydantic import BaseModel

app = Firecrawl(api_key="fc-YOUR-API-KEY")

class EventLinks(BaseModel):
    article: str
    tickets: str

class PageEvent(BaseModel):
    date: str
    day: str
    title: str
    performers: str
    works: str
    links: EventLinks

class PageEventList(BaseModel):
    events: list[PageEvent]

result = app.scrape(
    'https://www.gemeinde-gruenwald.de/kultur/terminkalender/index.html',
    formats=[{
        "type": "json",
        "schema": PageEventList.model_json_schema()
    }],
    include_tags=[".search-form__resultlist"]
)

print(result)
```
