```js Node
import Firecrawl from '@mendable/firecrawl-js';

const client = new Firecrawl({ apiKey: process.env.FIRECRAWL_API_KEY });

const doc = await client.scrape('https://firecrawl.dev', {
  formats: ['summary']
});

console.log(typeof doc.summary === 'string', (doc.summary || '').slice(0, 120));
```
